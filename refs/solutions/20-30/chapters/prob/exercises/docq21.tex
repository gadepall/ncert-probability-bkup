Let $Y_i \in \cbrak{1,2,3,4,5,6}$represent the outcome where $Y_1$ denotes black dice.

\begin{enumerate}
\item 
\begin{multline}
\pr{Y_1+Y_2 >9| Y_1 = 5} 
\\
= \frac{\pr{Y_1+Y_2 >9, Y_1 = 5}}{\pr{Y_1 = 5}}
\\
= \frac{\pr{Y_2 > 4, Y_1 = 5}}{\pr{Y_1 = 5}} 
\\
= \pr{Y_2 > 4} = \frac{1}{3}
\end{multline}
\item 
\begin{multline}
\pr{Y_1+Y_2 = 8| Y_2 < 4} 
\\
= \frac{\pr{Y_1 >4, Y_2 < 4}}{\pr{ Y_2 < 4}}
\\
=  \pr{Y_1 > 4} = \frac{1}{3}
\end{multline}
\end{enumerate}
