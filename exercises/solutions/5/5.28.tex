Probability of studying atleast 2 hours
\begin{align}
    &= P\brak{X\ge 2}\\
    &=\sum_{x=2}^4 P(X=x)\\
    &= P(X=2)+ P(X=3) +P(X=4)\\
    &= 2k + 2k +k\\
    &= 5 \times 0.15\\
    &= 0.75
\end{align}
Probability of studying exactly 2 hours
\begin{align}
    &=P(X=2)\\
    &= 0.15 \times 2\\
    &= 0.30
\end{align}
Probability of studying atmost 2 hours
\begin{align}
    &= P\brak{X\le 2}\\ 
    &=\sum_{x=0}^2 P(X=x)\\
    &= P(X=0) + P(X=1)+ P(X=2)\\
    &= 0.1 + 0.15 +0.30\\
