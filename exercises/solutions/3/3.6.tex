
\begin{align}\label{3.6:eq1}
   \pr{X=k} =
  \begin{cases*}
    {^n C_k}p^{k}(1-p)^{n-k} & $0 \leq k \leq n$\\
      0 & otherwise
  \end{cases*}
\end{align}
