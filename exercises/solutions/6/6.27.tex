\item 
\begin{align}
\pr{A + B} = \pr{A} + \pr{B} - \pr{AB}
\end{align}
Substituting the values,
\begin{align}
\frac{7}{11} = \frac{6}{11} + \frac{5}{11} - \pr{AB}\\
 \pr{AB} = 1 - \frac{7}{11} = \frac{4}{11}
\end{align}
\item
\begin{align}
\pr{A|B} = \frac{\pr{AB}}{\pr{B}} = \frac{4/11}{5/11} = \frac{4}{5}
\end{align}
\item
\begin{align}
\pr{B|A} = \frac{\pr{AB}}{\pr{A}} = \frac{4/11}{6/11} = \frac{2}{3}
\end{align}
\end{enumerate}
