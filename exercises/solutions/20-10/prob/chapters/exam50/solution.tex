Let
\begin{align}
X \epsilon \{1,2,3,4,5,6\}
\end{align}
For a fair dice,
\begin{align}
\pr{X=k}=\begin{cases}
        \frac{1}{6}& k=1,2,3,4,5,6\\
        0 & otherwise
        \end{cases}
\end{align}
\begin{enumerate}
\item 
\begin{align}
\pr{X > 4}=\pr{X=5}+\pr{X=6} = \frac{1}{3}
\end{align}
\item 
\begin{align}
\pr{X \leq 4}=1-\pr{X>4=}=\frac{2}{3}
\end{align}
\end{enumerate}
